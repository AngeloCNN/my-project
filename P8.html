from bs4 import BeautifulSoup
import time
url = 'https://mp.weixin.qq.com/s?__biz=MzI2MjM2MDEzNQ==&mid=2247487440&idx=1&sn=a4e15409d50ac5c2ecdf1d643ce1e3a2&chksm=ea4d12f8dd3a9beec8178b43f95b342ca9f4e2f0571f03872bdf491781c84a16afefd8e5f12e#rd'

    'DNT':'1',
'Host':'mp.weixin.qq.com',
'Origin':'https://mp.weixin.qq.com',
'Referer':'https://mp.weixin.qq.com/s?__biz=MzI2MjM2MDEzNQ==&mid=2247487440&idx=1&sn=a4e15409d50ac5c2ecdf1d643ce1e3a2&chksm=ea4d12f8dd3a9beec8178b43f95b342ca9f4e2f0571f03872bdf491781c84a16afefd8e5f12e',

'X-Requested-With':'XMLHttpRequest',
    'is_only_read':'1',
'req_id':'1409UzdQCkzabcoPnLjyfV6e',
'pass_ticket':'is_temp_url:0'
}
result = requests.post(url=url,headers=headers)
soup = BeautifulSoup(result.content,'lxml')
pic = soup.findAll('p',style="text-align: center;")
attr = []
for i in pic:
    img = i.find('img')
    try:
        a = img.attrs
        a = a['data-src']+'&'+a['data-src']+'&tp=webp&wxfrom=5&wx_lazy=1'
        #a = a['data-src']
        attr.append(a)
    except:
        continue
